    const processLabel = (label) => {
        if (label.length <= 14) return label;
        let words = label.split(' ');
        if(words.length === 1) {
            let lines = [];
            for(let i = 0; i < label.length; i+=12) {
                lines.push(label.substring(i, i+12));
            }
            return lines;
        }
        let lines = [];
        let currentLine = '';
        for(let i=0; i<words.length; i++) {
            if((currentLine + words[i]).length > 14) {
                if(currentLine.trim() !== '') lines.push(currentLine.trim());
                currentLine = words[i] + ' ';
            } else {
                currentLine += words[i] + ' ';
            }
        }
        if(currentLine.trim() !== '') lines.push(currentLine.trim());
        return lines.length > 1 ? lines : lines[0];
    };

    function openSwotDetail(type) {
        const detailView = document.getElementById('swotDetailView');
        const detailTitle = document.getElementById('detailTitle');
        const detailList = document.getElementById('detailList');
        const data = swotContent[type];
        
        detailTitle.textContent = data.title;
        detailTitle.className = 'text-4xl md:text-5xl font-black tracking-wide ' + data.colorClass;
        detailView.className = 'absolute inset-0 z-50 p-12 flex flex-col rounded-3xl border transition-opacity duration-300 backdrop-blur-2xl ' + data.bgClass;
        
        detailList.innerHTML = '';
        data.items.forEach(itemHtml => {
            const li = document.createElement('li');
            li.className = 'flex items-start';
            const bullet = document.createElement('span');
            bullet.className = 'mr-5 mt-3 w-4 h-4 rounded-full flex-shrink-0 shadow-[0_0_12px_currentColor] ' + data.bulletColor;
            const textSpan = document.createElement('div');
            textSpan.className = 'flex-1 leading-relaxed text-gray-200';
            textSpan.innerHTML = itemHtml;
            
            li.appendChild(bullet);
            li.appendChild(textSpan);
            detailList.appendChild(li);
        });
        
        detailView.classList.remove('hidden');
    }

    function closeSwotDetail() {
        document.getElementById('swotDetailView').classList.add('hidden');
    }
